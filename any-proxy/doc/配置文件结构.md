#配置文件格式和类型

```
is_tmp_file_io_page bool= true;

```

#跨平台格式：
```
全部平台都会去解析
name bool = value;
name usize = 131072;
只有window平台使用
name window_bool = value;
name window_usize = 131072;
只有linux, unix平台使用
name linux_bool = value;
name linux_usize = 131072;

name unix_bool = value;
name unix_usize = 131072;
```

#类型：
#单变量类型：
```
name bool = true;

name i8 = 0;
name i16 = 0;
name i32 = 0;
name isize = 0;
name i64 = 0;

name u8 = 0;
name u16 = 0;
name u32 = 0;
name usize = 0;
name u64 = 0;

name f64 = 0.0;

name str = 1234; 或 name str = "12   34";
```

#列表类型：
```
name i8s = 0 1 2;

name i64s = 0 1 2;

name u8s = 0 1 2; 
name u16s = 0 1 2;
name u32s = 0 1 2;
name usizes = 0 1 2;
name u64s = 0 1 2;

name f64s = 0.0 1.0 2.0;

name strs = 00 11 22; 或 name strs = "00 11 22"; 或 name strs = "0 0" "1 1" "2 2";
```

#u32, usize, u64, u32s, usizes, u64s
```
流量:
name u32 = 1;
name u32 = 1b;
name u32 = 1k;
name u32 = 1m;
name u32 = 1m;
name u32 = 1g;
names u32s = 1b 1k 1m;
时间:
name u32 = 1;  //1
name u32 = 1ms; //1
name u32 = 1s;  //1
name u32 = 0.1s;  // 0.1 * 1000
name u32 = 1min; //1 * 60
name u32 = 0.1min; //1 * 60 * 100

names u32s = 1ms 1min 0.1min;
浮点数是换算到毫秒,  非浮点数换算到秒
